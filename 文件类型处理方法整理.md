## 二、文本型文件

### 2.1 PDF（文字型）

**判断逻辑：**
```
.pdf 文件
    │
    ▼
PyMuPDF 提取第一页文字
    │
    ├─ 有文字（> 500字）→ 判定为文字型 PDF
    │
    └─ 无文字或极少
           ├─ 有内嵌图片 → 判定为扫描型 → Tesseract OCR
           └─ 无图片 → 记录为空白/损坏，状态=failed
```

**实现代码：**
```python
import fitz  # PyMuPDF

def extract_pdf_text(filepath: str) -> str:
    """提取 PDF 中所有文字（文字型 PDF）"""
    doc = fitz.open(filepath)
    results = []
    for i, page in enumerate(doc):
        text = page.get_text()
        if text.strip():
            results.append(f"--- 第{i+1}页 ---\n{text}")
    return "\n".join(results)

def is_scanned_pdf(filepath: str) -> bool:
    """判断 PDF 是否为扫描件（无文字层）"""
    doc = fitz.open(filepath)
    total_text_len = 0
    for page in doc:
        total_text_len += len(page.get_text().strip())
        if total_text_len > 500:
            return False  # 有文字，不是扫描件
        if page.get_images():
            return True   # 无文字但有图片，判定为扫描件
    return total_text_len == 0
```

**依赖：** `pip install pymupdf`

### 2.2 DOCX

```python
from docx import Document

def extract_docx(filepath: str) -> str:
    """提取 DOCX 中所有段落文字"""
    doc = Document(filepath)
    results = []
    for para in doc.paragraphs:
        if para.text.strip():
            results.append(para.text.strip())
    # 读表格
    for table in doc.tables:
        for row in table.rows:
            row_text = ' | '.join(cell.text.strip() for cell in row.cells if cell.text.strip())
            if row_text:
                results.append(row_text)
    return "\n".join(results)
```

**依赖：** `pip install python-docx`

### 2.3 PPTX

```python
from pptx import Presentation

def extract_pptx(filepath: str) -> str:
    """提取 PPTX 中所有文字，按幻灯片分组"""
    prs = Presentation(filepath)
    results = []
    for i, slide in enumerate(prs.slides):
        texts = []
        for shape in slide.shapes:
            if hasattr(shape, "text") and shape.text.strip():
                texts.append(shape.text.strip())
        if texts:
            results.append(f"=== Slide {i+1} ===\n" + "\n".join(texts))
    return "\n\n".join(results)
```

**依赖：** `pip install python-pptx`

### 2.4 MD / TXT

```python
def extract_text(filepath: str) -> str:
    """原生读取 MD / TXT"""
    encodings = ['utf-8', 'utf-8-sig', 'gbk', 'latin-1']
    for enc in encodings:
        try:
            with open(filepath, 'r', encoding=enc) as f:
                return f.read()
        except UnicodeDecodeError:
            continue
    return ""
```

### 2.5 HTML

```python
from bs4 import BeautifulSoup

def extract_html(filepath: str) -> str:
    """用 BeautifulSoup 提取 HTML 正文文本"""
    with open(filepath, 'r', encoding='utf-8', errors='ignore') as f:
        soup = BeautifulSoup(f.read(), 'html.parser')
    # 移除 script / style / nav / footer
    for tag in soup(['script', 'style', 'nav', 'footer', 'header']):
        tag.decompose()
    text = soup.get_text(separator='\n', strip=True)
    # 合并空行
    lines = [l for l in text.split('\n') if l.strip()]
    return '\n'.join(lines)
```

**依赖：** `pip install beautifulsoup4`

## 三、表格型文件

### 3.1 XLSX

```python
import openpyxl

def extract_xlsx(filepath: str) -> str:
    """提取 XLSX 全部单元格内容"""
    wb = openpyxl.load_workbook(filepath, data_only=True)
    results = []
    for sheet_name in wb.sheetnames:
        ws = wb[sheet_name]
        results.append(f"=== Sheet: {sheet_name} ===")
        for row in ws.iter_rows(values_only=True):
            row_text = ' | '.join(str(cell) if cell is not None else '' for cell in row)
            if row_text.strip():
                results.append(row_text)
    return "\n".join(results)
```

**依赖：** `pip install openpyxl`

### 3.2 XLS（老格式）

```python
import xlrd

def extract_xls(filepath: str) -> str:
    """提取老格式 XLS 全部单元格内容"""
    wb = xlrd.open_workbook(filepath)
    results = []
    for sheet_index in range(wb.nsheets):
        sheet = wb.sheet_by_index(sheet_index)
        results.append(f"=== Sheet: {sheet.name} ===")
        for row_index in range(sheet.nrows):
            row_text = ' | '.join(
                str(sheet.cell_value(row_index, col_index))
                for col_index in range(sheet.ncols)
            )
            if row_text.strip():
                results.append(row_text)
    return "\n".join(results)
```

**依赖：** `pip install xlrd`（注意：xlrd 2.0+ 仅支持 .xls，1.x 支持 .xls 和 .xlsx）

### 3.3 CSV

```python
import csv, codecs

def extract_csv(filepath: str) -> str:
    """提取 CSV 全部内容为文本"""
    encodings = ['utf-8-sig', 'gbk', 'utf-8', 'latin-1']
    for enc in encodings:
        try:
            with codecs.open(filepath, 'r', encoding=enc) as f:
                reader = csv.reader(f)
                results = []
                for row in reader:
                    row_text = ' | '.join(row)
                    if row_text.strip():
                        results.append(row_text)
                return '\n'.join(results)
        except UnicodeDecodeError:
            continue
    return ""
```

**依赖：** Python 标准库，无需安装

### 3.4 XLT（Excel 模板）

```python
import subprocess, os

def extract_xlt_via_conversion(filepath: str) -> str:
    """XLT 模板用 LibreOffice 转为 XLSX，再提取"""
    dest_dir = os.path.dirname(filepath)
    result = subprocess.run(
        ['soffice', '--headless', '--convert-to', 'xlsx',
         '--outdir', dest_dir, filepath],
        capture_output=True, text=True, timeout=120
    )
    new_path = filepath.replace('.xlt', '.xlsx')
    if os.path.exists(new_path):
        return extract_xlsx(new_path)
    return ""
```

### 3.5 NUMBERS（Apple Numbers）

```python
import zipfile, re

def extract_numbers(filepath: str) -> str:
    """Numbers 是 XML 打包的 zip，直接解析提取文字"""
    texts = []
    with zipfile.ZipFile(filepath, 'r') as zf:
        for name in zf.namelist():
            if name.endswith('.xml'):
                content = zf.read(name).decode('utf-8', errors='ignore')
                matches = re.findall(r'<a:t>([^<]+)</a:t>', content)
                texts.extend(matches)
    return '\n'.join(texts)
```

**依赖：** Python 标准库（zipfile + re）

```python
import subprocess

def ocr_image(image_path: str, lang: str = "chi_sim+eng") -> str:
    """调用 Tesseract 识别图片内文字"""
    result = subprocess.run(
        ["tesseract", image_path, "stdout", "-l", lang],
        capture_output=True, text=True, timeout=60

    return result.stdout

# 图片预处理（提升识别率）
```

```python
import subprocess, os

def doc_to_docx(filepath: str) -> str:
    """用 LibreOffice 将 .doc 转为 .docx"""
    dest_dir = os.path.dirname(filepath)
    result = subprocess.run(
        ['soffice', '--headless', '--convert-to', 'docx',
         '--outdir', dest_dir, filepath],
        capture_output=True, text=True, timeout=120
    )
    new_path = filepath + 'x'
    return new_path if os.path.exists(new_path) else ""

# 完整流程：转换后提取
def extract_doc(filepath: str) -> str:
    docx_path = doc_to_docx(filepath)
    if docx_path:
        return extract_docx(docx_path)
    return ""
```

## 七、处理流程总图

```
文件进入
    │
    ├─ .pdf  → PyMuPDF 提取文字 → 判定为文字型/扫描型
    │             ├─ 文字型（>500字）→ 直接入库
    │             └─ 扫描型 → Tesseract OCR
    │
    ├─ .docx → python-docx 提取
    ├─ .doc  → LibreOffice 转 .docx → python-docx 提取
    │
    ├─ .pptx → python-pptx 提取
    ├─ .ppt  → LibreOffice 转 .pptx → python-pptx 提取
    │
    ├─ .xlsx → openpyxl 提取
    ├─ .xls  → xlrd 提取
    ├─ .xlt  → LibreOffice 转 .xlsx → openpyxl 提取
    ├─ .csv  → csv 模块读取
    ├─ .numbers → zipfile 解析 XML 提取
    │
    ├─ .md / .txt → 原生读取
    ├─ .html → BeautifulSoup 提取
    │
    └─ .jpg / .png / .jpeg → Tesseract OCR
```
